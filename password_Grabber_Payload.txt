void type(int key , boolean release){
  Keyboard.press(key);
  if(release)
      Keyboard.release(key);
}

void toEnglish(){
  type(KEY_LEFT_SHIFT,false);
  type(KEY_LEFT_ALT,false);
  Keyboard.releaseAll();
  delay(100);
}


void print(const __FlashStringHelper *value){
